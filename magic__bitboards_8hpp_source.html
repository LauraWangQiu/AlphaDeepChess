237inline uint64_t magic_index_rook(uint64_t blockers, Square rook_square, uint64_t rook_attacks)
238{
239 // we cut unnecessary information to optimize memory space: board edges and squares outside the attack mask.
240 const uint64_t edges = ((ROW_1_MASK | ROW_8_MASK) & ~get_row_mask(rook_square.row())) |
241 ((COL_A_MASK | COL_H_MASK) & ~get_col_mask(rook_square.col()));
242 blockers &= rook_attacks & ~edges;
243
244#ifdef __BMI2__
245 const uint64_t occupancy_mask = rook_attacks & ~edges;
246 return _pext_u64(blockers, occupancy_mask); // Optimized version using pext if available.
247#else
248 // index is calculated by multiplying blockers by the magic number, then squeeze all bits to the right
249 return (blockers * rook_magic(rook_square)) >> (64 - ROOK_OCCUPANCY_NUMBER[rook_square]);
250#endif
251}
