173 static inline const std::array<std::array<int, 64>, 64> init_chebyshev_distance()
174 {
175 std::array<std::array<int, 64>, 64> distances;
176
177 for (Square sq1 = Square::A1; sq1.is_valid(); sq1++) {
178 for (Square sq2 = Square::A1; sq2.is_valid(); sq2++) {
179 distances[sq1][sq2] =
180 std::max(std::abs(int(sq1.row()) - int(sq2.row())), std::abs(int(sq1.col()) - int(sq2.col())));
181 }
182 }
183
184 return distances;
185 };

201 static inline const std::array<uint64_t, 64> init_king_danger_zone()
202 {
203 std::array<uint64_t, 64> zone;
204
205
206 for (Square sq = Square::A1; sq.is_valid(); sq++) {
207 const int king_row = sq.row();
208 const int king_col = sq.col();
209
210 // Calculate boundaries for a 3-square radius, clamped to the board
211 const int row_min = std::max(king_row - 3, 0);
212 const int row_max = std::min(king_row + 3, 7);
213 const int col_min = std::max(king_col - 3, 0);
214 const int col_max = std::min(king_col + 3, 7);
215
216 uint64_t mask = 0ULL;
217 // Iterate over the allowed rows and columns to set the mask bits
218 for (int row = row_min; row <= row_max; row++) {
219 // Get the full mask for the row, but then restrict it to allowed columns
220 uint64_t row_mask = get_row_mask(static_cast<Row>(row));
221 uint64_t col_range_mask = 0ULL;
222 for (int col = col_min; col <= col_max; col++) {
223 col_range_mask |= get_col_mask(static_cast<Col>(col));
224 }
225 mask |= row_mask & col_range_mask;
226 }
227 zone[sq] = mask;
228 }
229
230 return zone;
231 };
