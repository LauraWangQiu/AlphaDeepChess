436constexpr inline void Board::put_piece(Piece piece, Square square)
437{
438 assert(is_valid_piece(piece));
439 assert(square.is_valid());
440
441 if (piece == Piece::EMPTY) {
442 remove_piece(square);
443 return;
444 }
445
446 const ChessColor piece_color = get_color(piece);
447 const Piece previous_piece = array_piece[square];
448 const ChessColor previous_piece_color = get_color(previous_piece);
449
450 const uint64_t mask = square.mask();
451
452 // first remove the previous piece
453 bitboard_piece[static_cast<int>(previous_piece)] &= ~mask;
454 bitboard_color[static_cast<int>(previous_piece_color)] &= ~mask;
455
456 // place the new piece
457 bitboard_piece[static_cast<int>(piece)] |= mask;
