367constexpr inline PieceType piece_to_pieceType(Piece piece)
368{
369 assert(is_valid_piece(piece));
370
371 // the white and black pieces have a difference of +6 int value
372 // E.g WKnight = 1, BKnight = 7
373 // if piece is black then we substract 6 to convert to white.
374 const uint8_t type_int = static_cast<int>(piece);
375 return static_cast<PieceType>(type_int - (6 * (type_int >= 6)));
376}
